الأوسمة الأساسية في CSS

1. background-color & background-image

تستخدم لتعيين لون أو صورة خلفية العنصر.

background-color: #ffffff; /* لون خلفية أبيض */
background-image: url('image.jpg'); /* صورة خلفية */
background-repeat: no-repeat; /* منع تكرار الصورة */
background-size: cover; /* تغطية المساحة كاملة */

2. text

خصائص النص المختلفة.

text-align: center; /* محاذاة النص */
text-decoration: underline; /* تزيين النص */
text-transform: uppercase; /* تحويل النص */
line-height: 1.5; /* ارتفاع السطر */
letter-spacing: 2px; /* تباعد الحروف */

3. font

خصائص الخطوط.

font-family: 'Arial', sans-serif; /* نوع الخط */
font-size: 16px; /* حجم الخط */
font-weight: bold; /* سماكة الخط */
font-style: italic; /* نمط الخط */

4. color

لون النص.

color: #333333; /* لون النص */

5. border

حدود العنصر.

border: 1px solid #000000; /* حد بسمك 1px، صلب، لون أسود */
border-radius: 5px; /* زوايا مدورة */
border-top: 2px dashed red; /* حد علوي مختلف */

6. padding

المساحة الداخلية بين محتوى العنصر وحدوده.

padding: 10px; /* padding من جميع الجهات */
padding-top: 5px; /* padding من الأعلى فقط */
padding: 10px 20px; /* أعلى/أسفل 10px، يمين/يسار 20px */
padding: 5px 10px 15px 20px; /* أعلى، يمين، أسفل، يسار */

7. margin

المساحة الخارجية حول العنصر.

margin: 20px; /* margin من جميع الجهات */
margin-left: auto; /* margin تلقائي (مفيد للعناصر المركزة) */
margin: 0 auto; /* أعلى/أسفل 0، يمين/يسار تلقائي */

طرق كتابة CSS

1. CSS داخلي (Inline)

<p style="color: blue; font-size: 14px;">نص تجريبي</p>

2. CSS داخلي في <style>

<head>
<style>
p {
    color: red;
    font-family: Arial;
}
</style>
</head>

3. CSS خارجي (مستقل)

<head>
<link rel="stylesheet" href="styles.css">
</head>

4. CSS باستخدام @import

@import url('other-styles.css');

أمثلة عملية

مثال 1: زر بسيط

.button {
    background-color: #4CAF50;
    color: white;
    padding: 12px 24px;
    border: none;
    border-radius: 4px;
    font-size: 16px;
    margin: 10px;
    cursor: pointer;
}

.button:hover {
    background-color: #45a049;
}

جرب تمرير المؤشر فوق الزر لرؤية تأثير hover

مثال 2: بطاقة (Card)

.card {
    width: 300px;
    border: 1px solid #ddd;
    border-radius: 8px;
    padding: 16px;
    margin: 20px auto;
    box-shadow: 0 4px 8px rgba(0,0,0,0.1);
    background-color: white;
}

.card img {
    width: 100%;
    border-radius: 4px;
}

.card h3 {
    color: #333;
    font-size: 20px;
    margin-top: 10px;
}
